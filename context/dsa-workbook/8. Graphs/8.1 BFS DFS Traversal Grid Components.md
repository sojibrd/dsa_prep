```js
function numIslands(grid) {
  const m = grid.length,
    n = grid[0].length;
  let count = 0;
  const sink = (i, j) => {
    if (i < 0 || i >= m || j < 0 || j >= n || grid[i][j] !== "1") return;
    grid[i][j] = "0"; // visited = а¶°аІБа¶ђа¶ња¶ѓа¶ЉаІЗ а¶¶а¶ња¶®
    sink(i + 1, j);
    sink(i - 1, j);
    sink(i, j + 1);
    sink(i, j - 1);
  };
  for (let i = 0; i < m; i++)
    for (let j = 0; j < n; j++)
      if (grid[i][j] === "1") {
        count++;
        sink(i, j);
      }
  return count;
}
```
